# برنامه شمارش اعضای یک مجموعه در پایتون  # دریافت ورودی از کاربر user_input = input("یک مجموعه از اعداد یا رشته‌ها را با کاما وارد کنید: ")  # تبدیل ورودی به لیست items = user_input.split(",")  # تبدیل به مجموعه (برای حذف تکراری‌ها) my_set = set(item.strip() for item in items)  # چاپ تعداد اعضا print("تعداد اعضای مجموعه:", len(my_set))
